function addTask(phaseNumber, title) {
  setPhases(prev =>
    prev.map(p =>
      p.number === phaseNumber
        ? {
            ...p,
            tasks: [
              ...p.tasks,
              {
                id: Date.now(),
                title,
                status: "not_started",
                systemResource: null,
                userResource: null,
                note: ""
              }
            ]
          }
        : p
    )
  );
}

function updateTask(phaseNumber, taskId, updates) {
  setPhases(prev =>
    prev.map(p =>
      p.number === phaseNumber
        ? {
            ...p,
            tasks: p.tasks.map(t =>
              t.id === taskId ? { ...t, ...updates } : t
            )
          }
        : p
    )
  );
}
